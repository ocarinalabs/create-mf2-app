import { mutation, query } from "./_generated/server";
import { v } from "convex/values";

export const createReferralCode = mutation({
  handler: async (ctx) => {
    const user = await ctx.auth.getUserIdentity();
    if (!user) throw new Error("Not authenticated");
    
    const existing = await ctx.db
      .query("referralCodes")
      .withIndex("by_user", q => q.eq("userId", user.subject))
      .first();
    
    if (existing) return existing.code;
    
    const code = generateUniqueCode();
    await ctx.db.insert("referralCodes", {
      userId: user.subject,
      code,
      usageCount: 0,
      totalEarned: 0,
    });
    
    return code;
  },
});

### 3. AI understands your entire codebase

Because MF2 follows strict conventions:

```
AI knows:
- Pages live in src/app/*/page.tsx
- Components go in src/components/*
- Backend logic is in convex/*.ts
- Types are colocated with usage
- Imports use @/ prefix
```

So when you say "add a settings page", AI knows:
1. Create `src/app/settings/page.tsx`
2. Protect it with auth
3. Use existing UI components
4. Follow the established patterns

    - Use the same component patterns
    - Follow the same auth approach
    - Maintain consistency

## Setting up your AI environment

### Claude Code configuration

Add to your `CLAUDE.md`:

```markdown
# CLAUDE.md

## Project Context
This is an MF2 (Move F*cking Fast) stack application.

## Key Patterns
- Always use Convex for data (never local state for persisted data)
- UI components from shadcn/ui
- Real-time by default
- Server components where possible, "use client" when needed

## Commands
- `npm run dev` - Start Next.js
- `npx convex dev` - Start Convex (separate terminal)
- `npm run build` - Build for production
```

### Cursor rules

Create `.cursorrules`:

```
# MF2 Stack Rules

- Use TypeScript with strict mode
- Prefer Convex queries/mutations over API routes
- Components use shadcn/ui
- Follow Next.js 15 App Router patterns
- Real-time updates via useQuery hooks
- Authentication via Clerk
- Payments via Polar
- Emails via Resend
```

### GitHub Copilot workspace

`.github/copilot-workspace.md`:

```markdown
# MF2 Workspace

## Architecture
- Frontend: Next.js 15 + TypeScript + Tailwind
- Backend: Convex (real-time database)
- Auth: Clerk
- Payments: Polar

## Conventions
- Pages: src/app/*/page.tsx
- Components: src/components/*
- Backend: convex/*.ts
```
