### 3. Build the waitlist feature (3 minutes)

Create your waitlist schema:

```typescript convex/schema.ts
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  waitlist: defineTable({
    email: v.string(),
    paid: v.boolean(),
    earlyAccess: v.boolean(),
    createdAt: v.number(),
  })
    .index("by_email", ["email"])
    .index("by_paid", ["paid"]),
  
  // Keep existing tables...
});
```

Create the waitlist functions:

```typescript convex/waitlist.ts
import { v } from "convex/values";
import { mutation, query } from "./_generated/server";

export const join = mutation({
  args: { email: v.string() },
  handler: async (ctx, args) => {
    // Check if already on waitlist
    const existing = await ctx.db
      .query("waitlist")
      .withIndex("by_email", q => q.eq("email", args.email))
      .first();
    
    if (existing) {
      return { alreadyJoined: true, id: existing._id };
    }
    
    // Add to waitlist
    const id = await ctx.db.insert("waitlist", {
      email: args.email,
      paid: false,
      earlyAccess: false,
      createdAt: Date.now(),
    });
    
    return { alreadyJoined: false, id };
  },
});

export const getStats = query({
  handler: async (ctx) => {
    const total = await ctx.db.query("waitlist").collect();
    const paid = total.filter(u => u.paid).length;
    
    return {
      total: total.length,
      paid,
      conversion: total.length > 0 ? (paid / total.length) * 100 : 0,
    };
  },
});

export const markAsPaid = mutation({
  args: { waitlistId: v.id("waitlist") },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.waitlistId, {
      paid: true,
      earlyAccess: true,
    });
  },
});
```

### 4. Create the UI (2 minutes)

Replace the hero section in `src/app/page.tsx`:

```tsx src/app/page.tsx
"use client";

import { useState } from "react";
import { useMutation, useQuery } from "convex/react";
import { api } from "../convex/_generated/api";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";

export default function Home() {
  const [email, setEmail] = useState("");
  const [status, setStatus] = useState<"idle" | "loading" | "success">("idle");
  const join = useMutation(api.waitlist.join);
  const stats = useQuery(api.waitlist.getStats);

    e.preventDefault();
    setStatus("loading");

      // Redirect to payment
      window.location.href = `/pay?id=${result.id}`;

      setStatus("success");
      // Redirect to payment after 1 second
      setTimeout(() => {
        window.location.href = `/pay?id=${result.id}`;
      }, 1000);
    }
  };

  return (
    <div className="min-h-screen flex flex-col items-center justify-center p-4">

            Get early access to the course that teaches you how to build and launch
            a profitable SaaS in 48 hours.
          </p>
        </div>

        {stats && (
          <div className="flex justify-center gap-8 text-sm">
            <div>
              <div className="text-2xl font-bold">{stats.total}</div>
              <div className="text-muted-foreground">on waitlist</div>
            </div>
            <div>
              <div className="text-2xl font-bold">{stats.paid}</div>
              <div className="text-muted-foreground">paid for access</div>
            </div>

            <Input
              type="email"

              value={email}

              disabled={status !== "idle"}
              className="flex-1"

            <Button type="submit" disabled={status !== "idle"}>
```
